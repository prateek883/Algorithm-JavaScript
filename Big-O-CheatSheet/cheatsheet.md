Big O - Cheat sheet
Searching algorithms
# Binary search
TC -> O(logN)
SC -> O(1)
 
# Linear search
TC -> O(n)
SC -> O(1)


Sorting algorithms
# Big O | Sorting
 
# Bubble Sort
TC -> O(N^2)
SC -> O(1)
 
# Insertion Sort
TC -> O(N^2)
SC -> O(1)
 
# Selection Sort
TC -> O(N^2)
SC -> O(1)
 
# Merge Sort
TC -> O(NlogN) //base 2
SC -> O(n)
 
# Quick Sort
TC -> O(N^2)
SC -> O(logN) // base 2
 
# Count Sort
TC -> O(n+k) -> O(n)
SC -> O(n+k) -> O(n)
